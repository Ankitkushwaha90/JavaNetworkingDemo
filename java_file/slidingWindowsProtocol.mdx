import java.util.Scanner;

public class SlidingWindowProtocol {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input window size
        System.out.print("Enter the window size: ");
        int windowSize = sc.nextInt();

        // Input total number of frames
        System.out.print("Enter the total number of frames to send: ");
        int totalFrames = sc.nextInt();

        int framesSent = 0;

        while (framesSent < totalFrames) {
            int framesToSend = Math.min(windowSize, totalFrames - framesSent);

            System.out.println("\nSending frames: ");
            for (int i = 0; i < framesToSend; i++) {
                System.out.println("Frame " + (framesSent + i));
            }

            System.out.println("\nWaiting for Acknowledgement...");
            boolean allAcked = true;

            // Simulate Acknowledgement input
            for (int i = 0; i < framesToSend; i++) {
                System.out.print("Is Frame " + (framesSent + i) + " Acknowledged? (yes/no): ");
                String ack = sc.next();

                if (!ack.equalsIgnoreCase("yes")) {
                    System.out.println("Frame " + (framesSent + i) + " will be resent.");
                    totalFrames++; // Increase total frames to resend the lost frame
                    allAcked = false;
                }
            }

            if (allAcked) {
                framesSent += framesToSend;
            }
        }

        System.out.println("\nAll frames sent and acknowledged successfully!");

        sc.close();
    }
}
