```java
import java.util.HashMap;
import java.util.Scanner;

public class ARPSimulation {

    public static void main(String[] args) {
        // Step 1: Create a HashMap to store IP -> MAC mapping
        HashMap<String, String> arpTable = new HashMap<>();
        Scanner scanner = new Scanner(System.in);

        // Step 2: Get number of entries
        System.out.print("Enter the number of ARP table entries: ");
        int n = scanner.nextInt();
        scanner.nextLine(); // consume the leftover newline

        // Step 3: Input IP and MAC address pairs
        for (int i = 1; i <= n; i++) {
            System.out.print("Enter IP address " + i + ": ");
            String ip = scanner.nextLine();
            System.out.print("Enter MAC address " + i + ": ");
            String mac = scanner.nextLine();
            arpTable.put(ip, mac);
        }

        // Step 4: Display the ARP table
        System.out.println("\n--- ARP Table ---");
        for (String ip : arpTable.keySet()) {
            System.out.println("IP Address: " + ip + " -> MAC Address: " + arpTable.get(ip));
        }

        // Step 5: Search for a MAC address
        System.out.print("\nEnter IP address to search for MAC address: ");
        String searchIp = scanner.nextLine();

        // Step 6: Lookup and output
        if (arpTable.containsKey(searchIp)) {
            System.out.println("The MAC address for IP " + searchIp + " is: " + arpTable.get(searchIp));
        } else {
            System.out.println("MAC address for IP " + searchIp + " not found in ARP table.");
        }

        scanner.close();
    }
}
```
