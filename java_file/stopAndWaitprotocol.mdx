import java.util.Scanner;

public class StopAndWaitProtocol {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Input total number of frames
        System.out.print("Enter the number of frames to send: ");
        int totalFrames = sc.nextInt();

        int frameNumber = 0;

        while (frameNumber < totalFrames) {
            System.out.println("\nSending Frame " + frameNumber + "...");

            // Simulate waiting for Acknowledgement
            System.out.print("Was Frame " + frameNumber + " received successfully? (yes/no): ");
            String ack = sc.next();

            if (ack.equalsIgnoreCase("yes")) {
                System.out.println("Acknowledgment received for Frame " + frameNumber + ".");
                frameNumber++; // Move to next frame
            } else {
                System.out.println("Frame " + frameNumber + " lost! Resending Frame...");
                // Same frame will be resent (no increment)
            }
        }

        System.out.println("\nAll frames sent and acknowledged successfully!");
        sc.close();
    }
}
